# k8s-pki Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the external `kubecerts` Go tool with an in-repo, Nix-native Kubernetes PKI generator (`packages/k8s-pki`) that bootstraps, renews, and rotates the homelab cluster's PKI — all encrypted with agenix and committed to the `secrets/` submodule.

**Architecture:** Static PKI spec (`lib/k8s-pki/specs.nix`) + recipient policy (`lib/k8s-pki/recipients.nix`) are pulled into a derivation-baked `data.json` by `packages/k8s-pki/data.nix` (zero runtime `nix eval`). Four `writeShellApplication` subcommands (`bootstrap`/`renew`/`rotate-ca`/`status`) read that JSON and call `cfssl`/`age`/`openssl`. Agenix gains `hostPubkeys`/`userPubkeys` options; `mypki.nix` switches to a `common/` + per-host path layout with spec-driven assertions and `restartTriggers`.

**Tech Stack:** Nix (snowfall-lib), bash (`writeShellApplication`), `cfssl`, `age` (agenix), `openssl`, `jq`, `bats` (for shell helper tests). Target NixOS host: `m5p01`.

**Critical ordering constraint:** the m5p01 cluster must stay up throughout. Tasks 1–12 add new code with zero runtime effect; Task 13 introduces new `.age` files (old ones untouched); Task 14 flips `mypki.nix` to read the new files and rebuilds m5p01; Task 15 removes obsolete artifacts only after cluster health is verified.

**Flake+submodule gotcha:** this repo's `secrets/` is a git submodule. By default `nixos-rebuild --flake <path>` uses the `git+file:` URL scheme, which excludes submodule content — the agenix walker then returns empty and home-manager modules fail at eval. **Every `nixos-rebuild` / `nix build` / `nix run` command against this flake in this plan assumes `?submodules=1`** — e.g.:
- `nixos-rebuild build --flake '/persist/etc/nixos?submodules=1#m5p01'`
- `nix run '/persist/etc/nixos?submodules=1#k8s-pki.status'`
- `nix build '/persist/etc/nixos?submodules=1#k8s-pki.bootstrap'`

Add `?submodules=1` wherever you see a flake ref in this plan. `nix eval` with `--impure` via `builtins.getFlake "path:/persist/etc/nixos"` works without it (path: scheme copies everything), but that's not how the rebuild path resolves.

- [ ] **Step 1: Update walker to merge `secrets/common` + `secrets/<host>/system`**

Replace the `secrets = let ...` block with:
```nix
      secrets = let
        mkSecrets = prefix: files:
          foldl (acc: path:
            if snowfall.path.has-file-extension "age" path
            then acc // {
              ${removePrefix prefix (removeSuffix ".age" (builtins.unsafeDiscardStringContext path))}.file = path;
            }
            else acc) {} files;
        common-base = snowfall.fs.get-file "secrets/common";
        host-base = snowfall.fs.get-file "secrets/${host}/system";
        common-files =
          if builtins.pathExists common-base
          then snowfall.fs.get-files-recursive common-base
          else [];
        host-files =
          if builtins.pathExists host-base
          then snowfall.fs.get-files-recursive host-base
          else [];
      in
        (mkSecrets "${common-base}/" common-files) //
        (mkSecrets "${host-base}/"   host-files);
```

Note: key prefixes become `common/...` for files in `secrets/common/**` and stay the same for `secrets/<host>/system/**` (no change to existing keys).

### Task 4: Create `lib/k8s-pki/recipients.nix`

- [ ] **Step 1: Write the recipient-policy function**

Create `lib/k8s-pki/recipients.nix`:

```nix
# Pure recipient-policy function.
#
# Input:
#   hosts :: { <name> = { name; role; advertiseIP; masterAddress; hostPubkeys; }; ... }
#   users :: { <name@host> = { name; host; userPubkeys; }; ... }
#
# Output:
#   { <relativePath> = [ "ssh-ed25519 AAAA..." ... ]; ... }
#
# Relative paths are the same keys used in age.secrets (no ".age" suffix).

{ lib }:
{ hosts, users }:
let
  inherit (lib) flatten concatMap attrValues filterAttrs mapAttrs' nameValuePair unique;

  allHostPubkeys   = flatten (concatMap (h: h.hostPubkeys)   (attrValues hosts));
  masterPubkeys    = flatten (concatMap (h: h.hostPubkeys)
    (attrValues (filterAttrs (_: h: h.role == "master") hosts)));
  allUserPubkeys   = flatten (concatMap (u: u.userPubkeys)   (attrValues users));

  publicRecipients  = unique (allHostPubkeys ++ allUserPubkeys);
  privateRecipients = unique (masterPubkeys  ++ allUserPubkeys);

  commonPublic = [
    "common/k8s-pki/ca.crt"
    "common/k8s-pki/etcd/ca.crt"
    "common/k8s-pki/front-proxy-ca.crt"
    "common/k8s-pki/sa.pub"
  ];

  commonPrivate = [
    "common/k8s-pki/ca.key"
    "common/k8s-pki/etcd/ca.key"
    "common/k8s-pki/front-proxy-ca.key"
    "common/k8s-pki/sa.key"
  ];

  common =
    builtins.listToAttrs (map (p: { name = p; value = publicRecipients;  }) commonPublic)
    //
    builtins.listToAttrs (map (p: { name = p; value = privateRecipients; }) commonPrivate);

  perHost = lib.concatMapAttrs (hostName: h:
    let
      hostRecipients = unique (h.hostPubkeys ++ allUserPubkeys);
      # per-host leaves live under homelab-k8s/... for agenix keys on the host
      # but in the secrets/ tree under secrets/<host>/system/homelab-k8s/...
      prefix = "secrets/${hostName}/system/homelab-k8s";
    in
    builtins.listToAttrs (map (leaf: {
      name = "${prefix}/${leaf}";
      value = hostRecipients;
    }) [
      "apiserver.crt" "apiserver.key"
      "apiserver-etcd-client.crt" "apiserver-etcd-client.key"
      "apiserver-kubelet-client.crt" "apiserver-kubelet-client.key"
      "front-proxy-client.crt" "front-proxy-client.key"
      "controller-manager.crt" "controller-manager.key"
      "controller-manager-client.crt" "controller-manager-client.key"
      "scheduler.crt" "scheduler.key"
      "scheduler-client.crt" "scheduler-client.key"
      "addon-manager-client.crt" "addon-manager-client.key"
      "cluster-admin-client.crt" "cluster-admin-client.key"
      "kubelet.crt" "kubelet.key"
      "kubelet-client.crt" "kubelet-client.key"
      "kube-proxy-client.crt" "kube-proxy-client.key"
      "flannel-client.crt" "flannel-client.key"
      "flannel-etcd-client.crt" "flannel-etcd-client.key"
      "etcd/server.crt" "etcd/server.key"
      "etcd/peer.crt" "etcd/peer.key"
      "etcd/healthcheck-client.crt" "etcd/healthcheck-client.key"
    ])
  ) hosts;

  perUser = lib.concatMapAttrs (_: u:
    let prefix = "secrets/${u.host}/home/${u.name}/homelab-k8s";
    in builtins.listToAttrs (map (f: {
      name = "${prefix}/${f}";
      value = unique u.userPubkeys;
    }) [
      "ca.crt"
      "cluster-admin.crt"
      "cluster-admin.key"
    ])
  ) users;
in
  # Keys in the common block use agenix-style relative paths;
  # keys in perHost/perUser use full repo-relative paths under secrets/.
  # Merge: the shell script branches on key prefix ("common/" vs "secrets/").
  common // perHost // perUser
```

```bash
# Shared helpers for k8s-pki shell commands.
# Intended to be sourced by cmd-*.sh; also exercised by tests/lib.bats.

set -euo pipefail

# die MSG [CODE] — print error and exit
die() {
  local msg="$1"
  local code="${2:-1}"
  printf '%s\n' "error: $msg" >&2
  exit "$code"
}

# log LEVEL MSG — stderr log line (levels: info, warn, err)
log() {
  local level="$1"; shift
  printf '[%s] %s\n' "$level" "$*" >&2
}

# pem_expiry_seconds PATH — seconds from now until notAfter (negative if expired)
pem_expiry_seconds() {
  local path="$1"
  local not_after now
  not_after=$(openssl x509 -in "$path" -noout -enddate | sed 's/^notAfter=//')
  now=$(date -u +%s)
  local exp_epoch
  exp_epoch=$(date -u -d "$not_after" +%s)
  echo $((exp_epoch - now))
}

# age_recipients_of PATH — print recipient header lines ("ssh-ed25519 <hash>", etc.)
# Reads only the age header (plaintext header prefix before "--- " separator).
age_recipients_of() {
  local path="$1"
  # age header ends with a line starting with "--- "; recipients are lines starting "-> "
  awk '
    /^--- / { exit }
    /^-> / { sub(/^-> /, ""); print }
  ' "$path"
}

# recipient_set_eq A B — exit 0 if A and B (newline-separated) represent the same set
recipient_set_eq() {
  local a b
  a=$(printf '%s\n' "$1" | sort -u)
  b=$(printf '%s\n' "$2" | sort -u)
  [ "$a" = "$b" ]
}

# age_encrypt_to FILE_OUT < stdin — encrypt stdin to env $RECIPIENTS (newline-separated pubkeys)
age_encrypt_to() {
  local out="$1"
  local args=()
  while IFS= read -r r; do
    [ -z "$r" ] && continue
    args+=(-r "$r")
  done <<<"${RECIPIENTS:-}"
  age "${args[@]}" -o "$out"
}

# age_decrypt_in FILE — decrypt ciphertext to stdout using default age identity discovery
age_decrypt_in() {
  local in="$1"
  age -d "$in"
}

# jq_data EXPR — run jq over $K8S_PKI_DATA
jq_data() {
  jq -r "$1" "$K8S_PKI_DATA"
}

# cfssl_gen_ca NAME CN EXPIRY OUTDIR — generate a CA keypair; writes $OUTDIR/$NAME.{pem,key}
cfssl_gen_ca() {
  local name="$1" cn="$2" expiry="$3" outdir="$4"
  local csr
  csr=$(mktemp)
  jq -n --arg cn "$cn" --arg expiry "$expiry" '{
    CN: $cn,
    key: { algo: "rsa", size: 2048 },
    ca: { expiry: $expiry }
  }' > "$csr"
  cfssl gencert -initca "$csr" | cfssljson -bare "$outdir/$name"
  rm -f "$csr" "$outdir/$name.csr"
}

# cfssl_gen_leaf NAME CN CA_PREFIX PROFILE_JSON HOSTS_JSON EXPIRY OUTDIR
# Generates a leaf cert signed by CA at $CA_PREFIX.{pem,-key.pem}.
# Writes $OUTDIR/$NAME.{pem,key} and removes the intermediate .csr.
cfssl_gen_leaf() {
  local name="$1" cn="$2" ca_prefix="$3" profile_json="$4" hosts_json="$5" expiry="$6" outdir="$7"
  local csr cfg
  csr=$(mktemp); cfg=$(mktemp)
  jq -n --arg cn "$cn" --argjson hosts "$hosts_json" '{
    CN: $cn,
    hosts: $hosts,
    key: { algo: "rsa", size: 2048 }
  }' > "$csr"
  jq -n --argjson profile "$profile_json" --arg expiry "$expiry" '{
    signing: {
      default: { expiry: $expiry },
      profiles: { default: ($profile + { expiry: $expiry }) }
    }
  }' > "$cfg"
  cfssl gencert \
    -ca="$ca_prefix.pem" -ca-key="$ca_prefix-key.pem" \
    -config="$cfg" -profile=default "$csr" \
    | cfssljson -bare "$outdir/$name"
  rm -f "$csr" "$cfg" "$outdir/$name.csr"
}
```

```bash
# Decide action for a given policy key and existing ciphertext.
# Prints one of: GENERATE | REENCRYPT | RESIGN | SKIP
decide_action() {
  local key="$1" ct="$2"
  local actual expected

  [ -f "$ct" ] || { echo GENERATE; return; }

  if [ "$FORCE" -eq 1 ] && [[ "$key" != common/* || "$key" == common/k8s-pki/sa.* ]]; then
    : # FORCE only applies to leaves; CAs skipped here
  fi

  if [ "$FORCE" -eq 1 ] && [[ "$key" != common/* ]]; then
    echo RESIGN; return
  fi

  case "$key" in
    *.crt)
      local pt pemfile secs days
      pt=$(age_decrypt_in "$ct" 2>/dev/null) || { echo GENERATE; return; }
      pemfile=$(mktemp); echo "$pt" > "$pemfile"
      secs=$(pem_expiry_seconds "$pemfile" 2>/dev/null || echo 0)

      if [ "$days" -lt 30 ] && [[ "$key" != common/* ]]; then
        echo RESIGN; return
      fi ;;
  esac

  expected=$(recips_for "$key" | sort -u)
  actual=$(age_recipients_of "$ct" | sort -u)

  [ "$(echo "$expected" | grep -c .)" -eq "$(echo "$actual" | grep -c .)" ] \

  echo SKIP

# PHASE 1 — CAs and SA
gen_cas() {
  local ca
  for ca in $(jq -r '.specs.cas | keys[]' "$K8S_PKI_DATA"); do
    local crt_key="common/k8s-pki/${ca}.crt"
    local key_key="common/k8s-pki/${ca}.key"

      log info "GENERATE CA $ca"
      [ "$DRY_RUN" -eq 1 ] && continue

    else
      for pair in "crt:$action_crt:$crt_key:$ct_crt" "key:$action_key:$key_key:$ct_key"; do
        IFS=: read -r _kind action policy ct <<<"$pair"
        case "$action" in
          REENCRYPT)
            log info "RE-ENCRYPT $policy"
            [ "$DRY_RUN" -eq 1 ] && continue
            age_decrypt_in "$ct" | RECIPIENTS=$(recips_for "$policy") age_encrypt_to "${ct}.new"
            mv "${ct}.new" "$ct"
            ;;
          SKIP) log info "skip (current) $policy" ;;
          *) ;;  # GENERATE handled above jointly; RESIGN doesn't apply to CAs
        esac
      done
    fi
  done

  # SA keypair — plain RSA, no x509
  local sa_pub_key="common/k8s-pki/sa.pub"
  local sa_key_key="common/k8s-pki/sa.key"
  local sa_pub_ct; sa_pub_ct=$(ct_path_for "$sa_pub_key")
  local sa_key_ct; sa_key_ct=$(ct_path_for "$sa_key_key")

  if [ ! -f "$sa_pub_ct" ] || [ ! -f "$sa_key_ct" ]; then
    log info "GENERATE SA keypair"
    [ "$DRY_RUN" -eq 1 ] || {
      local bits; bits=$(jq -r '.specs.sa.keyBits' "$K8S_PKI_DATA")
      local tmpdir; tmpdir=$(mktemp -d)
      openssl genrsa -out "$tmpdir/sa.key" "$bits" 2>/dev/null
      openssl rsa   -in  "$tmpdir/sa.key" -pubout -out "$tmpdir/sa.pub" 2>/dev/null
      mkdir -p "$(dirname "$sa_pub_ct")"
      RECIPIENTS=$(recips_for "$sa_pub_key") age_encrypt_to "$sa_pub_ct" < "$tmpdir/sa.pub"
      RECIPIENTS=$(recips_for "$sa_key_key") age_encrypt_to "$sa_key_ct" < "$tmpdir/sa.key"
      rm -rf "$tmpdir"
    }
  else
    for pair in "pub:$sa_pub_key:$sa_pub_ct" "key:$sa_key_key:$sa_key_ct"; do
      IFS=: read -r _kind policy ct <<<"$pair"
      action=$(decide_action "$policy" "$ct")
      case "$action" in
        REENCRYPT)
          log info "RE-ENCRYPT $policy"
          [ "$DRY_RUN" -eq 1 ] && continue
          age_decrypt_in "$ct" | RECIPIENTS=$(recips_for "$policy") age_encrypt_to "${ct}.new"
          mv "${ct}.new" "$ct"
          ;;
        SKIP) log info "skip (current) $policy" ;;
      esac
    done
  fi
}

      action_key=$(decide_action "$key_key" "$ct_key")

          mkdir -p "$(dirname "$outdir/${leaf}")"
          cfssl_gen_leaf "$leaf" "$cn" "$ca_prefix" "$profile_json" "$hosts_json" "$expiry" "$outdir"
          mkdir -p "$(dirname "$ct_crt")"
          RECIPIENTS=$(recips_for "$crt_key") age_encrypt_to "$ct_crt" < "$outdir/${leaf}.pem"
          RECIPIENTS=$(recips_for "$key_key") age_encrypt_to "$ct_key" < "$outdir/${leaf}-key.pem"
          rm -rf "$outdir"
          ;;
        REENCRYPT:*|*:REENCRYPT)
          for pair in "crt:$action_crt:$crt_key:$ct_crt" "key:$action_key:$key_key:$ct_key"; do

            mv "${ct}.new" "$ct"
```

```bash
      local scope; scope=$(jq -r --arg l "$leaf" '.specs.leaves[$l].scope' "$K8S_PKI_DATA")
      local role; role=$(jq -r --arg h "$h" '.hosts[$h].role' "$K8S_PKI_DATA")
      [ "$scope" = "all" ] || [ "$scope" = "$role" ] || continue
      local crt key
      crt=$(ct_path_for "secrets/${h}/system/homelab-k8s/${leaf}.crt")
      key=$(ct_path_for "secrets/${h}/system/homelab-k8s/${leaf}.key")
      [ "$DRY_RUN" -eq 1 ] && { log info "would delete $crt $key"; continue; }
      rm -f "$crt" "$key"
    done
  done
  # user certs (only for ca)
  if [ "$ca" = "ca" ]; then
    jq -r '.users | to_entries[] | "\(.value.host)\t\(.value.name)"' "$K8S_PKI_DATA" \
      | while IFS=$'\t' read -r uhost uname; do
        for f in cluster-admin.crt cluster-admin.key ca.crt; do
          local p; p=$(ct_path_for "secrets/${uhost}/home/${uname}/homelab-k8s/${f}")
          [ "$DRY_RUN" -eq 1 ] && { log info "would delete $p"; continue; }
          rm -f "$p"
        done
      done
  fi
}

if [ -z "$CA_NAME" ]; then
  for ca in $(jq -r '.specs.cas | keys[]' "$K8S_PKI_DATA"); do
    delete_ca "$ca"
    delete_leaves_for_ca "$ca"
  done
  # SA keypair
  if [ "$DRY_RUN" -ne 1 ]; then
    rm -f "$(ct_path_for common/k8s-pki/sa.pub)" "$(ct_path_for common/k8s-pki/sa.key)"
    log info "deleted SA keypair ciphertexts"
  fi
else
  jq -e --arg ca "$CA_NAME" '.specs.cas[$ca]' "$K8S_PKI_DATA" >/dev/null \
    || die "unknown CA: $CA_NAME" 2
  delete_ca "$CA_NAME"
  delete_leaves_for_ca "$CA_NAME"
fi

if [ "$DRY_RUN" -eq 1 ]; then
  log info "dry run complete — no files changed"
  exit 0
fi

# Invoke bootstrap to regenerate everything that's now missing
log info "invoking bootstrap to regenerate deleted material"
bash -c "$(
  cat <<'INNER'
export K8S_PKI_LIB K8S_PKI_DATA
source "$K8S_PKI_LIB"
INNER
)
$(cat "$(dirname "$K8S_PKI_LIB")/cmd-bootstrap.sh")" || die "bootstrap failed during rotate-ca" 1

log info "rotate-ca complete — remember: nixos-rebuild switch, kubeconfig re-trust"
```

Note: the chained `bash -c` invocation assumes `cmd-bootstrap.sh` is alongside `lib.sh` in the same directory. Adjust in `packages/k8s-pki/default.nix` (Task 12) to bundle both files so `dirname "$K8S_PKI_LIB"` contains both. Alternative cleaner implementation: export `K8S_PKI_BOOTSTRAP=${./scripts/cmd-bootstrap.sh}` alongside `K8S_PKI_LIB`, then `bash "$K8S_PKI_BOOTSTRAP"`.
